\section{Budapest}
\begin{frame}{Budapest}
    \begin{itemize}
        \item Budapest is the capital of Hungary, known for its stunning architecture and vibrant culture.
        \item The city is divided by the Danube River and features many historical landmarks.
    \end{itemize}
\end{frame}

\section{Tokaj}
\begin{frame}{Tokaj}
    \begin{itemize}
        \item Tokaj is a historic wine region in Hungary, famous for its white wines.
        \item Known for the Tokaji Aszú, a sweet wine produced from grapes affected by noble rot.
        \item The region is a UNESCO World Heritage site, celebrated for its unique vineyards and wine cellars.
    \end{itemize}
\end{frame}
